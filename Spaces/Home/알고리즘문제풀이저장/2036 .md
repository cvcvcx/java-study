import java.util.Scanner;  
  
class Solution{  
    public static void main(String[] args){  
    //10으로 나눈 나머지를 계속해서 더하면  
    //각 자릿수의 합을 모두 더할 수 있음  
    //더하는 순서에 주의할것  
    Scanner s = new Scanner(System.in);  
        int a = s.nextInt();  
        int result = 0;  
        while (a/10>0){  
            result += a%10;  
            a /= 10;  
        }        result += a;  
        System.out.println(result);  
    }}
